cmake_minimum_required(VERSION 3.10)

project(btc-tracker)

find_package(fmt CONFIG REQUIRED)
find_package(cpp-httplib CONFIG REQUIRED)
find_package(nlohmann_json CONFIG REQUIRED)

add_executable(btc-tracker main.cpp)

target_link_libraries(btc_tracker PRIVATE fmt::fmt cpp-httplib::cpp-httplib nlohmann_json::nlohmann_json)
